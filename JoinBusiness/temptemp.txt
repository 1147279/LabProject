select WEATHER.ID, WEATHER.YEAR, WEATHER.MONTH, WEATHER.DAY, WEATHER.HOUR, WEATHER.MINUTE, WEATHER.SECOND, WEATHER.LOCATION, WEATHER.TEMPERATURE, HUMIDITY.YEAR, HUMIDITY.MONTH, HUMIDITY.DAY, HUMIDITY.HOUR, HUMIDITY.MINUTE, HUMIDITY.SECOND, HUMIDITY.LOCATION, HUMIDITY.HUMIDITY FROM WEATHER INNER JOIN HUMIDITY ON WEATHER.YEAR = HUMIDITY.YEAR and WEATHER.MONTH = HUMIDITY.MONTH and WEATHER.DAY = HUMIDITY.DAY and WEATHER.HOUR = HUMIDITY.HOUR and WEATHER.MINUTE = HUMIDITY.MINUTE and WEATHER.SECOND = HUMIDITY.SECOND and WEATHER.LOCATION = HUMIDITY.LOCATION; > bbbb.csv
